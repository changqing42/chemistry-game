function initThree(){

  const canvas=document.getElementById('mol-canvas');

  const wrap=document.getElementById('canvas-wrap');

  renderer=new THREE.WebGLRenderer({canvas,antialias:true,alpha:true});

  renderer.setPixelRatio(Math.min(window.devicePixelRatio,2));

  renderer.setClearColor(0x000000,0);

  resizeR();


  scene=new THREE.Scene();

  camera=new THREE.PerspectiveCamera(38,wrap.clientWidth/wrap.clientHeight,0.1,200);

  camera.position.set(0,0,13);


  const al=new THREE.AmbientLight(0xffffff,0.5); scene.add(al);

  const dl=new THREE.DirectionalLight(0xffffff,0.9); dl.position.set(5,8,6); scene.add(dl);

  const dl2=new THREE.DirectionalLight(0x3366ff,0.35); dl2.position.set(-4,-3,-5); scene.add(dl2);


  molGroup=new THREE.Group(); scene.add(molGroup);


  // --- PC 端鼠标事件 ---

  canvas.addEventListener('mousedown',e=>{isDrag=true;prevM={x:e.clientX,y:e.clientY};});

  window.addEventListener('mouseup',()=>isDrag=false);

  window.addEventListener('mousemove',e=>{

    if(!isDrag)return;

    rotY+=(e.clientX-prevM.x)*0.009; rotX+=(e.clientY-prevM.y)*0.009;

    prevM={x:e.clientX,y:e.clientY};

  });

  canvas.addEventListener('wheel',e=>{

    camera.position.z=Math.max(5,Math.min(22,camera.position.z+e.deltaY*0.018));

  },{passive:true});

  canvas.addEventListener('click', e => triggerRaycast(e.clientX, e.clientY));


  // --- 移动端触控事件适配 ---

  canvas.addEventListener('touchstart', e => {

    if(e.touches.length === 1){

      isDrag = true;

      touchStartX = e.touches[0].clientX;

      touchStartY = e.touches[0].clientY;

      prevM = {x: touchStartX, y: touchStartY};

    }

  }, {passive: false});

  window.addEventListener('touchend', () => isDrag=false);

  

  canvas.addEventListener('touchmove', e => {

    if(!isDrag || e.touches.length !== 1) return;

    e.preventDefault(); // 阻止浏览器滚动

    rotY += (e.touches[0].clientX - prevM.x) * 0.008;

    rotX += (e.touches[0].clientY - prevM.y) * 0.008;

    prevM = {x: e.touches[0].clientX, y: e.touches[0].clientY};

  }, {passive: false});

  

  // 移动端点击防误触判定

  canvas.addEventListener('touchend', e => {

    if(e.changedTouches.length === 1){

      let dx = e.changedTouches[0].clientX - touchStartX;

      let dy = e.changedTouches[0].clientY - touchStartY;

      if(Math.abs(dx) < 8 && Math.abs(dy) < 8){ // 未发生明显拖动，判定为点击

        triggerRaycast(e.changedTouches[0].clientX, e.changedTouches[0].clientY);

      }

    }

  });


  window.addEventListener('resize', resizeR);

  requestAnimationFrame(loop);

}


function resizeR(){

  const w=document.getElementById('canvas-wrap');

  renderer.setSize(w.clientWidth,w.clientHeight,false);

  if(camera){camera.aspect=w.clientWidth/w.clientHeight;camera.updateProjectionMatrix();}

}


function loop(){

  requestAnimationFrame(loop);

  if(molGroup){ molGroup.rotation.x=rotX; molGroup.rotation.y=rotY; }

  renderer.render(scene,camera);

}


function resetAllMeshColors() {

  clickableAtoms.forEach(a => {

    a.mesh.material.emissive.setHex(0x000000);

    a.mesh.material.emissiveIntensity = 0;

    a.mesh.scale.setScalar(1.0);

  });

  clickableBonds.forEach(b => {

    if(b.mesh.material.transparent && b.mesh.material.opacity === 0) return;

    b.mesh.material.emissive.setHex(0x000000);

    b.mesh.material.emissiveIntensity = 0;

    b.mesh.scale.setScalar(1.0);

  });

}


let globalBondLines=[]; 

function buildMolecule(key){

  while(molGroup.children.length) molGroup.remove(molGroup.children[0]);

  clickableAtoms=[]; clickableBonds=[]; bondMeshMap={}; atomMeshMap={}; globalBondLines=[];

  const mol=MOLS[key]; if(!mol)return;


  let cx=0,cy=0,cz=0;

  mol.atoms.forEach(a=>{cx+=a.p[0];cy+=a.p[1];cz+=a.p[2];});

  cx/=mol.atoms.length;cy/=mol.atoms.length;cz/=mol.atoms.length;


  mol.bonds.forEach((b,bi)=>{

    const cnt=b.tp==='triple'?3:b.tp==='double'?2:1;

    for(let si=0;si<cnt;si++){ globalBondLines.push({mainBondIdx:bi,subIdx:si,fromAtom:b.f,toAtom:b.t,type:b.tp}); }

  });


  mol.atoms.forEach((a,ai)=>{

    const r=a.r*0.46;

    const geo=new THREE.SphereGeometry(r,28,28);

    const mat=new THREE.MeshPhongMaterial({color:a.c, shininess:90, specular:0x333333});

    const mesh=new THREE.Mesh(geo,mat);

    mesh.position.set(a.p[0]-cx,a.p[1]-cy,a.p[2]-cz);

    mesh.userData={type:'atom',atomIdx:ai};

    molGroup.add(mesh);

    clickableAtoms.push({mesh,atomIdx:ai});

    atomMeshMap[ai]=mesh;

  });


  globalBondLines.forEach((line,gbi)=>{

    const p1=new THREE.Vector3(mol.atoms[line.fromAtom].p[0]-cx,mol.atoms[line.fromAtom].p[1]-cy,mol.atoms[line.fromAtom].p[2]-cz);

    const p2=new THREE.Vector3(mol.atoms[line.toAtom].p[0]-cx,mol.atoms[line.toAtom].p[1]-cy,mol.atoms[line.toAtom].p[2]-cz);

    const meshes=drawBondLine(p1,p2,line.type,line.subIdx,gbi);

    bondMeshMap[gbi]=meshes;

    meshes.forEach(m=>{

      m.userData={type:'bond',globalBondIdx:gbi,mainBondIdx:line.mainBondIdx,subIdx:line.subIdx,bondType:line.type};

      clickableBonds.push({mesh:m,globalBondIdx:gbi,mainBondIdx:line.mainBondIdx});

    });

  });


  if(mol.benzeneRing) addBenzeneCloud();

  resetAllMeshColors();

}

  const radii={single:0.075,double:0.058,triple:0.050,aromatic:0.065};

  const offsets={single:0,double:subIdx===0?0.13:-0.13,triple:subIdx===0?0:subIdx===1?0.14:-0.14,aromatic:0};


  const geo=new THREE.CylinderGeometry(radii[type],radii[type],len*0.78,12);

  const mat=new THREE.MeshPhongMaterial({color:colors[type],shininess:60});

  const mesh=new THREE.Mesh(geo,mat);


  const hitR=0.20; // 稍微增加移动端点击范围

  const hitGeo=new THREE.CylinderGeometry(hitR,hitR,len*0.85,8);

  const hitMat=new THREE.MeshBasicMaterial({transparent:true,opacity:0,depthWrite:false});

  const hitMesh=new THREE.Mesh(hitGeo,hitMat);


  const offsetVec=perp.clone().multiplyScalar(offsets[type]);

  [mesh,hitMesh].forEach(m=>{

    m.position.copy(mid).add(offsetVec);

    const ax=new THREE.Vector3(0,1,0);

    m.quaternion.setFromUnitVectors(ax,dir.clone().normalize());

  });


  molGroup.add(mesh); molGroup.add(hitMesh);

  hitMesh.userData={type:'bond',globalBondIdx:gbi,mainBondIdx:gbi,subIdx,bondType:type};

  clickableBonds.push({mesh:hitMesh,globalBondIdx:gbi,mainBondIdx:gbi});

  return [mesh,hitMesh];

}

const raycaster=new THREE.Raycaster();

const mouse2d=new THREE.Vector2();


function triggerRaycast(clientX, clientY) {

  if(!currentMolKey||!currentQ)return;

  const rect=renderer.domElement.getBoundingClientRect();

  mouse2d.x=((clientX-rect.left)/rect.width)*2-1;

  mouse2d.y=-((clientY-rect.top)/rect.height)*2+1;

  raycaster.setFromCamera(mouse2d,camera);


  const allMeshes=[...clickableAtoms.map(x=>x.mesh), ...clickableBonds.map(x=>x.mesh)];

  const hits=raycaster.intersectObjects(allMeshes);

  if(!hits.length)return;


  const obj=hits[0].object;

  const ud=obj.userData;

  const resDiv = document.getElementById(`qresult-${currentQ.qIdx}`);

  if(resDiv) { resDiv.className='q-result'; resDiv.innerHTML=''; }


  if(ud.type==='atom') handleAtomClick(ud.atomIdx);

  else if(ud.type==='bond') handleBondClick(ud.globalBondIdx,ud.mainBondIdx);

}


function handleAtomClick(atomIdx){

  const q=getCurrentQ();

  if(!q||q.done || q.mode==='bond')return; 

  if(selectedAtoms.has(atomIdx)){ selectedAtoms.delete(atomIdx); setAtomHighlight(atomIdx,false); } 

  else { selectedAtoms.add(atomIdx); setAtomHighlight(atomIdx,true); }

  updateSelectionDisplay();

}


function handleBondClick(gbi,mainBondIdx){

  if(selectedBonds.has(key)){ selectedBonds.delete(key); setBondHighlight(gbi,false); }

  updateSelectionDisplay();

}


function setAtomHighlight(atomIdx,on){

  const mesh=atomMeshMap[atomIdx]; if(!mesh)return;

  if(on){ mesh.material.emissive.setHex(0x00aaff); mesh.material.emissiveIntensity=0.7; mesh.scale.setScalar(1.22); } 

  else { mesh.material.emissive.setHex(0x000000); mesh.material.emissiveIntensity=0; mesh.scale.setScalar(1.0); }

}


function setBondHighlight(gbi,on){

  const meshes=bondMeshMap[gbi]; if(!meshes)return;

  meshes.forEach(m=>{

    if(m.material.transparent&&m.material.opacity===0)return;

    else { m.material.emissive.setHex(0x000000); m.material.emissiveIntensity=0; m.scale.setScalar(1.0); }

}

function checkAnswer(q, atomSel, bondSel){

  const mol=MOLS[currentMolKey];

  if(q.mode==='bond'){

    if(q.correct.bondIdx){

      const slots=q.correct.bondIdx; let ok=true;

      let uniqueSelectedMainBonds = new Set();

      Array.from(bondSel).forEach(k=>{ const gbi=parseInt(k.replace('b','')); uniqueSelectedMainBonds.add(globalBondLines[gbi]?globalBondLines[gbi].mainBondIdx:undefined); });

      if(uniqueSelectedMainBonds.size !== slots.length) return false;

      slots.forEach(slot=>{ const hit = Array.from(uniqueSelectedMainBonds).some(mainBi => slot.includes(mainBi)); if(!hit) ok=false; });

      return ok;

    }

    if(q.correct.bondTypes){

      let uniqueMainBonds = new Set();

      Array.from(bondSel).forEach(k=>{ const gbi=parseInt(k.replace('b','')); uniqueMainBonds.add(globalBondLines[gbi]?globalBondLines[gbi].mainBondIdx:undefined); });

      if(uniqueMainBonds.size !== q.count) return false;

      return Array.from(uniqueMainBonds).every(mainBi => {

         const line = globalBondLines.find(l => l.mainBondIdx === mainBi); if(!line) return false;

         const fa=mol.atoms[line.fromAtom].s, ta=mol.atoms[line.toAtom].s;

         const pair=`${fa}-${ta}`, pairR=`${ta}-${fa}`;

         return q.correct.bondTypes.some(bt=>{

           if(bt==='C-H')return (pair==='C-H'||pairR==='C-H')&&line.type==='single';

           if(bt==='O-H')return (pair==='O-H'||pairR==='O-H')&&line.type==='single';

           if(bt==='C=C')return line.type==='double';

           if(bt==='C≡C')return line.type==='triple';

           return false;

         });

      });

    }

  }

  if(q.mode==='multi'){

    const ca=q.correct.atomIdx||[]; const cb=q.correct.bondIdx||[];

    const atomOk = ca.every(ai=>atomSel.has(ai)) && atomSel.size===ca.length;

    const selMainBonds = [...new Set(Array.from(bondSel).map(k=>{

      const gbi=parseInt(k.replace('b','')); return globalBondLines[gbi]?globalBondLines[gbi].mainBondIdx:undefined;

    }).filter(v=>v!==undefined))];

    const bondOk = cb.every(bi=>selMainBonds.includes(bi)) && selMainBonds.length===cb.length;

    return atomOk && bondOk;

  }

  return false;

}

        const fa=mol.atoms[line.fromAtom].s,ta=mol.atoms[line.toAtom].s; const pair=`${fa}-${ta}`, pairR=`${ta}-${fa}`;

        const match=q.correct.bondTypes.some(bt=>{ if(bt==='C-H')return (pair==='C-H'||pairR==='C-H')&&line.type==='single'; if(bt==='O-H')return (pair==='O-H'||pairR==='O-H')&&line.type==='single'; if(bt==='C=C')return line.type==='double'; if(bt==='C≡C')return line.type==='triple'; return false; });
